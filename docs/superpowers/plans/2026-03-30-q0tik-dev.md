# q0tik.dev Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a pixel-art personal portfolio site with an animated cat mascot on the navbar, deployed to GitHub Pages.

**Architecture:** Single-page static site (index.html + CSS + JS). Animated pixel-art cat generated via PixelLab MCP, rendered as CSS sprite-sheet animation with JS state machine. No build tools or frameworks.

**Tech Stack:** HTML5, CSS3, vanilla JS, Google Fonts (Press Start 2P), PixelLab MCP for asset generation, GitHub Pages for hosting.

Queue animations for other states. Check available quadruped templates first via `get_character`. Likely available: `idle`, `sleep`, `eat`, `attack`.

- Blink: custom `action_description`: "sitting and blinking eyes"
- Head turn: custom `action_description`: "sitting and turning head to look around"
- Grooming: use `eat` template or custom "sitting and grooming, licking paw"

- [ ] **Step 5: Download all animation frames and assemble sprite sheet**

- [ ] **Step 2: Write cat animation state machine in js/cat.js**

```js
// Cat sprite animation state machine
(function () {
  const container = document.getElementById('cat');
  if (!container) return;

  const sprite = document.createElement('div');
  sprite.className = 'cat-sprite idle';
  container.appendChild(sprite);

  const ACTIONS = [
    { name: 'blink', className: 'blink', duration: 300, minInterval: 3000, maxInterval: 6000 },
    { name: 'head-turn', className: 'head-turn', duration: 800, minInterval: 8000, maxInterval: 15000 },
    { name: 'groom', className: 'groom', duration: 1500, minInterval: 15000, maxInterval: 30000 },
  ];

  let isActing = false;

  function randomBetween(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
  }

  function playAction(action) {
    if (isActing) return;
    isActing = true;

    sprite.className = 'cat-sprite ' + action.className;

    setTimeout(function () {
      sprite.className = 'cat-sprite idle';
      isActing = false;
    }, action.duration);
  }

  function scheduleAction(action) {
    var delay = randomBetween(action.minInterval, action.maxInterval);
    setTimeout(function () {
      playAction(action);
      scheduleAction(action);
    }, delay);
  }

  ACTIONS.forEach(function (action) {
    scheduleAction(action);
  });
})();
```

```js
// Burger menu toggle
(function () {
  var burger = document.getElementById('burger');
  var navLinks = document.querySelector('.nav-links');

  if (burger && navLinks) {
    burger.addEventListener('click', function () {
      navLinks.classList.toggle('open');
      burger.classList.toggle('open');
    });

    // Close menu on link click
    navLinks.querySelectorAll('.nav-link').forEach(function (link) {
      link.addEventListener('click', function () {
        navLinks.classList.remove('open');
        burger.classList.remove('open');
      });
    });
  }
})();

// Scroll reveal
(function () {
  var sections = document.querySelectorAll('.section');

  var observer = new IntersectionObserver(function (entries) {
    entries.forEach(function (entry) {
      if (entry.isIntersecting) {
        entry.target.classList.add('visible');
      }
    });
  }, { threshold: 0.15 });

  sections.forEach(function (section) {
    observer.observe(section);
  });
})();

// Active nav link on scroll
(function () {
  var navLinks = document.querySelectorAll('.nav-link');
  var sections = document.querySelectorAll('.section');

  var observer = new IntersectionObserver(function (entries) {
    entries.forEach(function (entry) {
      if (entry.isIntersecting) {
        var id = entry.target.id;
        navLinks.forEach(function (link) {
          link.classList.toggle('active', link.getAttribute('href') === '#' + id);
        });
      }
    });
  }, { threshold: 0.3 });

  sections.forEach(function (section) {
    observer.observe(section);
  });
})();
```
